**File:** `src/components/diary/RodPickerSheet.tsx`

### Today

Mounts as a `<Sheet>` (bottom drawer). Works on mobile; on desktop the bottom-sheet pattern is unconventional and can feel cramped.

### Change

Detect desktop viewport via `window.matchMedia("(min-width: 768px)")` (or use the project's existing breakpoint hook if there is one — check `src/hooks/use-mobile.tsx` or similar).

Branch:
- **Mobile** — keep the existing `<Sheet>` rendering.
- **Desktop** — render the same content as a centered modal `<Dialog>` with `max-w-[480px]` instead of a bottom sheet.

Both paths render the same internal component (rod list + setup-new-rod CTA). Just different shells.

```tsx
import { useEffect, useState } from "react";

function useIsDesktop() {
  const [isDesktop, setIsDesktop] = useState(
    typeof window !== "undefined" && window.matchMedia("(min-width: 768px)").matches
  );
  useEffect(() => {
    const mq = window.matchMedia("(min-width: 768px)");
    const handler = (e: MediaQueryListEvent) => setIsDesktop(e.matches);
    mq.addEventListener("change", handler);
    return () => mq.removeEventListener("change", handler);
  }, []);
  return isDesktop;
}

// In RodPickerSheet:
const isDesktop = useIsDesktop();
return isDesktop ? <DialogShell>{body}</DialogShell> : <SheetShell>{body}</SheetShell>;
```

Where `<DialogShell>` wraps the existing children in a centered `<Dialog>` with `<DialogContent>`.

## §4 — 16-oz overflow guard in CatchFlow

**File:** `src/components/diary/CatchFlow.tsx`

### Today

Inside `handleSave`:

```ts
weight_lb = Math.floor(f);
weight_oz = Math.round((f - weight_lb) * 16);
```

For `f = 2.99`: `Math.floor(2.99) = 2`, `Math.round((0.99) * 16) = 16`. Result: `weight_lb = 2, weight_oz = 16`. The display becomes "2 lb 16 oz" instead of "3 lb 0 oz".

### Change

After the calculation, normalise:

```ts
weight_lb = Math.floor(f);
weight_oz = Math.round((f - weight_lb) * 16);
if (weight_oz >= 16) {
  weight_lb += 1;
  weight_oz = 0;
}
```

### Verification

- Type `2.99` lb. Save. Confirm DB row has `weight_lb = 3, weight_oz = 0, weight_display = "3 lb"`.
- Type `2.5` lb. Confirm `weight_lb = 2, weight_oz = 8, weight_display = "2 lb 8 oz"`. (Regression check.)
- Type `1.0` lb. Confirm `weight_lb = 1, weight_oz = 0`. (Regression check.)

#### 6.2 Create `VenuePickerOverlay.tsx`

Mirrors `AskGhillieOverlay.tsx`'s pattern — a phase-mounted full-page picker:

```tsx
interface Props {
  sessionId: string;
  currentVenueName: string;
  onClose: () => void;
  onSwitched: (newVenueName: string, newVenueType: "stillwater" | "river" | null) => void;
}

export default function VenuePickerOverlay({ sessionId, currentVenueName, onClose, onSwitched }: Props) {
  const [search, setSearch] = useState("");
  const [venues, setVenues] = useState<VenueOption[]>([]);
  // Reuse the loader pattern from DiaryNew — load venues_new + classify water type
  // Filter by search input
  // On pick: update fishing_sessions.venue_name (+ venue_type if known) + emit Change event

  async function handlePick(venue: VenueOption) {
    const { error } = await supabase
      .from("fishing_sessions")
      .update({ venue_name: venue.name, venue_type: venue.waterType ?? undefined })
      .eq("id", sessionId);
    if (error) { toast.error(error.message); return; }

    // Emit a Change event so the journal reflects the switch
    await supabase.from("session_events").insert({
      session_id: sessionId,
      event_type: "change",
      event_time: new Date().toISOString(),
      change_type: "venue",
      change_from: currentVenueName,
      change_to: venue.name,
      change_reason: "mid-session venue switch",
    });

    logEvent("venue.switched", { session_id: sessionId, from: currentVenueName, to: venue.name });
    onSwitched(venue.name, venue.waterType);
  }

  return (
    <div className="max-w-[440px] mx-auto p-4 space-y-4 pb-32">
      <div className="flex items-center gap-3">
        <button onClick={onClose} aria-label="Close" className="p-2 -ml-2">
          <ArrowLeft className="h-5 w-5" />
        </button>
        <h2 className="text-lg font-semibold font-diary">Switch venue</h2>
      </div>

      <Input
        placeholder="Filter venues…"
        value={search}
        onChange={(e) => setSearch(e.target.value)}
      />

      <ul className="space-y-1">
        {filteredVenues.map((v) => (
          <li key={v.name}>
            <button
              type="button"
              onClick={() => handlePick(v)}
              className="w-full text-left rounded-md border px-3 py-2 hover:bg-muted/40"
              disabled={v.name === currentVenueName}
            >
              <div className="text-sm font-medium">{v.name}</div>
              <div className="text-xs text-muted-foreground">{v.waterType ?? "—"}</div>
            </button>
          </li>
        ))}
      </ul>
    </div>
  );
}
```
